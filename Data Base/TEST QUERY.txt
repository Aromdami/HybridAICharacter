CREATE TABLE IF NOT EXISTS 교수
(	
	pName VARCHAR(20), ID VARCHAR2(4) NOT NULL, 

	CONSTRAINT prf_PK 
		PRIMARY KEY(ID)
);


CREATE TABLE IF NOT EXISTS 수업 
(
	ID VARCHAR(4), Name VARCHAR2(30), pID VARCHAR2(4), cRoom NUMBER(4), cDay1 NUMBER(1), cDay2 NUMBER(1), startTime TIME, endTime TIME,

	CONSTRAINT class_PK 
		PRIMARY KEY (ID),
	CONSTRAINT class_FK 
		FOREIGN KEY (pID)
		REFERENCES 교수(ID)
);

[각 질의응답]

1. 모든 수업을 보여줘, 올해 수업을 보여줘, 이 건물에 무슨 수업이 있어?, 에러 등

SELECT name, pname, croom, cday1, cday2, startTime, endTime
FROM 수업 classes
INNER JOIN 교수 profs
	ON classes.pid = profs.id
ORDER BY cday1 ASC, startTime ASC


2. 오늘 있는 수업을 보여줘

SELECT name, pname, croom, cday1, cday2, startTime, endTime
FROM 수업 classes
INNER JOIN 교수 profs
	ON classes.pid = profs.id
WHERE cday1 = strftime('%w', 'now', 'localtime') or cday2 = strftime('%w', 'now', 'localtime')
ORDER BY startTime ASC


3. ㅁㅁㅁ 교수님 수업이 뭐 있는지 알려줘

SELECT name, croom, cday1, cday2, startTime, endTime
FROM 수업 classes
INNER JOIN 교수 profs
	ON classes.pid = profs.id
WHERE profs.pname = "우균"
ORDER BY cday1 ASC, startTime ASC


4. 지금 N호실 수업 뭐 하고 있어? N호실 비어 있어?
(수업 종료 시간 데이터 추가해야함)

SELECT name, pname, croom, cday1, cday2, startTime, endTime
FROM 수업 classes
INNER JOIN 교수 profs
	ON classes.pid = profs.id
WHERE (cday1 = strftime('%w', 'now', 'localtime') or cday2 = strftime('%w', 'now', 'localtime')) AND (strftime('%H:%M', 'now', 'localtime') > startTime AND strftime('%H:%M', 'now', 'localtime') < endTime)
ORDER BY cday1 ASC, startTime ASC



5. ㅇㅇㅇ 수업 어디서 해?

FROM 수업 classes

ORDER BY cday1 ASC, startTime ASC
