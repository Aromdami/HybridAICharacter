CREATE TABLE 교수
(	
	pName VARCHAR(20), pID VARCHAR2(4) NOT NULL, 

	CONSTRAINT prf_PK 
		PRIMARY KEY(pID)
);


CREATE TABLE 수업 
(
	cID VARCHAR(4), cName VARCHAR2(30), pID VARCHAR2(4), cRoom NUMBER(4), cDay1 NUMBER(1), cDay2 NUMBER(1), cTime TIME,

	CONSTRAINT class_PK 
		PRIMARY KEY (cID),
	CONSTRAINT class_FK 
		FOREIGN KEY (pID)
		REFERENCES 교수(pID)
);

[각 질의응답]

1. 모든 수업을 보여줘, 올해 수업을 보여줘, 이 건물에 무슨 수업이 있어?, 에러 등

SELECT cname, pname, croom, cday1, cday2, ctime
FROM 수업 classes
INNER JOIN 교수 profs
	ON classes.pid = profs.pid
ORDER BY cday1 ASC, ctime ASC


2. 오늘 있는 수업을 보여줘

SELECT cname, pname, croom, cday1, cday2, ctime
FROM 수업 classes
INNER JOIN 교수 profs
	ON classes.pid = profs.pid
WHERE cday1 = strftime('%w', CURRENT_DATE) or cday2 = strftime('%w', CURRENT_DATE)
ORDER BY ctime ASC


3. ㅁㅁㅁ 교수님 수업이 뭐 있는지 알려줘

SELECT cname, croom, cday1, cday2, ctime
FROM 수업 classes
INNER JOIN 교수 profs
	ON classes.pid = profs.pid
WHERE profs.pname = "ㅁㅁㅁ"
ORDER BY cday1 ASC, ctime ASC

5. ㅇㅇㅇ 수업 어디서 해?
